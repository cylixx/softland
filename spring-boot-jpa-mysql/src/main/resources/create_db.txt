CREATE TABLE
    TEST.user
    (
        id BIGINT NOT NULL AUTO_INCREMENT,
        folio NUMERIC,
        name VARCHAR(100),
        address VARCHAR(120),
        birthday DATE,
        datereg DATETIME,
        CONSTRAINT id_key PRIMARY KEY (id),
        CONSTRAINT user_idx_id UNIQUE (id),
        CONSTRAINT user_idx_folio UNIQUE (folio)
    )
    ENGINE=InnoDB DEFAULT CHARSET=utf8;
